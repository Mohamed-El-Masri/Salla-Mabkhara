⚡ كود JavaScript (ضعه في قسم JavaScript مخصص)

function addIncenseToRatings() {
    const ratingElements = document.querySelectorAll('.comment__rating');
    
    ratingElements.forEach(function(ratingElement) {
        if (!ratingElement.querySelector('.incense-icon-custom')) {
            const incenseIcon = document.createElement('span');
            incenseIcon.className = 'incense-icon-custom';
            
            setTimeout(function() {
                incenseIcon.classList.add('loaded');
            }, 100);
            
            ratingElement.insertBefore(incenseIcon, ratingElement.firstChild);
        }
    });
}

document.addEventListener('DOMContentLoaded', addIncenseToRatings);
window.addEventListener('load', addIncenseToRatings);

const incenseObserver = new MutationObserver(function(mutations) {
    let shouldUpdate = false;
    
    mutations.forEach(function(mutation) {
        mutation.addedNodes.forEach(function(node) {
            if (node.nodeType === Node.ELEMENT_NODE) {
                if (node.querySelector && node.querySelector('.comment__rating')) {
                    shouldUpdate = true;
                } else if (node.classList && node.classList.contains('comment__rating')) {
                    shouldUpdate = true;
                }
            }
        });
    });
    
    if (shouldUpdate) {
        setTimeout(addIncenseToRatings, 100);
    }
});

incenseObserver.observe(document.body, {
    childList: true,
    subtree: true
});

window.updateIncenseIcons = addIncenseToRatings;
setInterval(addIncenseToRatings, 5000);
